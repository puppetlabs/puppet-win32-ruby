def inheritanceOrderClasses(cls)
  sortArray = cls.dup
  i1 = 0
  while i1 < sortArray.size-1
    again = false
    for i2 in i1+1..sortArray.size-1
      e2 = sortArray[i2]
      if sortArray[i1].eSuperTypes.include?(e2)
        sortArray.delete(e2)
        sortArray.insert(i1,e2)
        again = true
        break
      end
    end
    i1 += 1 unless again
  end
  sortArray
end
