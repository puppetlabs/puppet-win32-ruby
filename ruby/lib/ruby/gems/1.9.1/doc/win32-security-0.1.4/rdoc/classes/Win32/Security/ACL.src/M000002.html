      def initialize(revision = ACL_REVISION)
        acl = 0.chr * 8 # This can be increased later as needed

        unless InitializeAcl(acl, acl.size, revision)
          raise Error, get_last_error
        end

        @acl = acl
        @revision = revision
      end
