        def GetLargestConsoleWindowSize( hConsoleOutput )
          @GetLargestConsoleWindowSize ||= Win32API.new( "kernel32", "GetLargestConsoleWindowSize", ['l'], 'l' )
          coord = @GetLargestConsoleWindowSize.call( hConsoleOutput )
          x = coord >> 16
          y = coord & 0x0000ffff
          return [x,y]
        end
