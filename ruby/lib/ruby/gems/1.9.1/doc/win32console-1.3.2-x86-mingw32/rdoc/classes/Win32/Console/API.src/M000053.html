       def GenerateConsoleCtrlEvent( dwCtrlEvent, dwProcessGroupId )
         @GenerateConsoleCtrlEvent ||= Win32API.new( "kernel32", "GenerateConsoleCtrlEvent", ['l', 'l'], 'l' )
         @GenerateConsoleCtrlEvent.call( dwCtrlEvent, dwProcessGroupId )
       end
