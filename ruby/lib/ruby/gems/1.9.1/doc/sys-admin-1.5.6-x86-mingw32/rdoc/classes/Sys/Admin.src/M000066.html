    def self.get_login
      buffer = 0.chr * 256
      nsize  = [buffer.size].pack("L")

      if GetUserName.call(buffer, nsize) == 0
        raise Error, 'GetUserName() call failed in get_login'
      end

      length   = nsize.unpack('L')[0]
      username = buffer[0 ... length].chop
      username
    end
